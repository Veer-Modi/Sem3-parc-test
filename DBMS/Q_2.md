## Question-2

```sql

INSERT INTO MenuItems (item_id, item_name, price, category)
VALUE
(1,'Sandwich',150,'Fast Food'),
(2,'Pizza',250,'Main Courses'),
(3,'Ice Cream',50,'Dessert');


UPDATE MenuItems
SET price = 200
WHERE item_id = 2;


DELETE FROM MenuItems
WHERE item_id = 3;


SELECT AVG(price) AS Average_Price
FROM MenuItems;


SELECT SUM(price) AS Total_Price
FROM MenuItems;
```
